
void setup()
{
  pinMode(8, OUTPUT);
  pinMode(9, OUTPUT);
  
  pinMode(3, OUTPUT);
  pinMode(4, OUTPUT);
  pinMode(5, OUTPUT);
  pinMode(6, OUTPUT);
  
}

void cycle()
{
	for(int i=8;i<=9;i++)
    {
      digitalWrite(i,1);
  	
    for(int j=3;j<7;j++)
    {
    	digitalWrite(j,1);
      	delay(300);
      	digitalWrite(j,0);
                
    }
   	digitalWrite(i,0);
      
    }
}

void loop()
{
  
  cycle();
  
  

}
